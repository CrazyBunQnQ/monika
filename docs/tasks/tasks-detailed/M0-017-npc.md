## NPC 结构

```typescript
interface NPC {
  // === 标识 ===
  id: string;
  /** NPC 唯一标识符 */

  name: string;
  /** NPC 名称 */

  // === 基本信息 ===
  appearance: NPCAppearance;
  /** 外貌描述 */

  personality: NPCPersonality;
  /** 性格特征 */

  background?: string;
  /** 背景故事 */

  // === 属性 (可选，战斗 NPC) ===
  attributes?: NPCAttributes;

  derived?: NPCDerivedStats;
  /** 派生数值 */

  // === 技能 ===
  skills?: Record<string, number>;
  /** 技能值列表 */

  // === 战斗数据 ===
  combat?: NPCCombatData;

  // === 状态 ===
  state?: NPCState;

  // === 关系 ===
  relationships?: NPCRelationship[];

  // === 元数据 ===
  role: NPCRole;
  /** NPC 在故事中的角色 */

  importance: 'minor' | 'supporting' | 'major' | 'key';
  /** 重要程度 */

  is_unique?: boolean;
  /** 是否是独特 NPC (不可重复) */

  // === KP 信息 ===
  notes?: string;
  /** KP 备注 (仅 KP 可见) */

  secrets?: string[];
  /** 秘密信息 (仅 KP 可见) */

  // === 扩展 ===
  custom_data?: Record<string, any>;
  /** 自定义数据 */
}

type NPCRole =
  | 'witness'          // 证人
  | 'suspect'          // 嫌疑人
  | 'victim'           // 受害者
  | 'ally'             // 盟友
  | 'enemy'            // 敌人
  | 'neutral'          // 中立
  | 'merchant'         // 商人
  | 'authority'        // 权威人物
  | 'expert'           // 专家
  | 'monster';         // 怪物
```

## NPCAppearance 结构

```typescript
interface NPCAppearance {
  age?: number | string;
  /** 年龄，可以是数字或范围 (如 "30-40") */

  gender?: string;
  /** 性别 */

  height?: string;
  /** 身高 */

  build?: string;
  /** 体型，如 "瘦弱", "中等", "魁梧" */

  hair?: string;
  /** 发色和发型 */

  eyes?: string;
  /** 眼睛颜色和特征 */

  distinctive_features?: string[];
  /** 明显特征，如疤痕、纹身等 */

  clothing?: string;
  /** 穿着打扮 */

  mannerisms?: string[];
  /** 行为习惯、动作特征 */

  first_impression?: string;
  /** 第一印象 */

  voice?: string;
  /** 声音特征 */
}
```

## NPCPersonality 结构

```typescript
interface NPCPersonality {
  traits?: string[];
  /** 性格特质，如 ["好奇", "谨慎", "贪婪"] */

  motivations?: string[];
  /** 动机，如 ["寻求真相", "保护家人"] */

  fears?: string[];
  /** 恐惧，如 ["黑暗", "孤独"] */

  goals?: string[];
  /** 目标 */

  quirks?: string[];
  /** 怪癖、特殊习惯 */

  attitudes?: {
    /** 对不同事物的态度 */
    towards_investigators?: string;
    towards_supernatural?: string;
    towards_authority?: string;
  };

  speech_pattern?: string;
  /** 说话方式，如 "正式", "口吃", "方言" */
}
```

## NPCAttributes 结构

```typescript
interface NPCAttributes {
  STR?: number;  // 力量
  CON?: number;  // 体质
  DEX?: number;  // 敏捷
  APP?: number;  // 外貌
  POW?: number;  // 意志
  INT?: number;  // 智力
  SIZ?: number;  // 体型
  EDU?: number;  // 教育
}

interface NPCDerivedStats {
  HP?: number;
  HP_max?: number;

  MP?: number;
  MP_max?: number;

  SAN?: number;
  SAN_max?: number;

  Luck?: number;
  Luck_max?: number;

  Move?: number;
  /** 移动速度 */

  Build?: string;
  /** 体格: "弱小", "普通", "强壮", "巨大" */

  DamageBonus?: string;
  /** 伤害加值，如 "-1d4", "+1d6" */
}
```

## NPCCombatData 结构

```typescript
interface NPCCombatData {
  attacks?: NPCAttack[];
  /** 攻击方式列表 */

  armor?: number;
  /** 护甲值 */

  evasion?: number;
  /** 闪避值 */

  block?: number;
  /** 格挡值 */

  combat_manuevers?: string[];
  /** 战斗技巧 */

  special_abilities?: NPCSpecialAbility[];
  /** 特殊能力 */

  weapons?: string[];
  /** 携带武器 */

  loot?: string[];
  /** 掉落物品 */
}

interface NPCAttack {
  name: string;
  /** 攻击名称 */

  type: 'melee' | 'ranged';
  /** 攻击类型 */

  accuracy: number;
  /** 命中值 */

  damage: string;
  /** 伤害，如 "1d6+DB" */

  description?: string;
  /** 描述 */

  special?: string;
  /** 特殊效果 */
}

interface NPCSpecialAbility {
  name: string;
  /** 能力名称 */

  description: string;
  /** 描述 */

  trigger?: string;
  /** 触发条件 */

  effect?: string;
  /** 效果 */

  usage?: string;
  /** 使用限制，如 "每回合1次" */
}
```

## NPCState 结构

```typescript
interface NPCState {
  alive?: boolean;
  /** 是否存活 */

  conscious?: boolean;
  /** 是否清醒 */

  location?: string;
  /** 当前位置 */

  condition?: string;
  /** 当前状态，如 "健康", "受伤", "恐惧", "敌对" */

  hp?: number;
  /** 当前 HP */

  san?: number;
  /** 当前 SAN */

  knowledge?: {
    /** 知道的信息 */
    knows_clues?: string[];
    knows_secrets?: string[];
  };

  attitude_towards?: {
    /** 对他人的态度 */
    investigators?: 'friendly' | 'neutral' | 'hostile' | 'afraid';
  };
}
```
