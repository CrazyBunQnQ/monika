## 后端国际化

```python
# app/core/i18n.py
from typing import Dict, Optional
import json
from pathlib import Path

class I18nService:
    """国际化服务"""

    def __init__(self):
        self.current_lang = "zh"
        self.translations: Dict[str, Dict[str, str]] = {}
        self._load_translations()

    def _load_translations(self):
        """加载翻译文件"""
        locales_dir = Path("app/locales")

        for locale_file in locales_dir.glob("*.json"):
            lang_code = locale_file.stem
            with open(locale_file, 'r', encoding='utf-8') as f:
                self.translations[lang_code] = json.load(f)

    def t(
        self,
        key: str,
        lang: Optional[str] = None,
        **kwargs
    ) -> str:
        """翻译文本"""
        lang = lang or self.current_lang

        if lang not in self.translations:
            lang = "zh"  # 默认中文

        translation = self.translations[lang].get(key, key)

        # 格式化参数
        if kwargs:
            try:
                translation = translation.format(**kwargs)
            except (KeyError, ValueError):
                pass

        return translation

    def set_language(self, lang: str):
        """设置当前语言"""
        if lang in self.translations:
            self.current_lang = lang

    def get_available_languages(self) -> list:
        """获取可用语言列表"""
        return [
            {"code": "zh", "name": "简体中文"},
            {"code": "en", "name": "English"},
            {"code": "ja", "name": "日本語"},
        ]

# 全局实例
_i18n = I18nService()

def t(key: str, **kwargs) -> str:
    """翻译快捷函数"""
    return _i18n.t(key, **kwargs)

def set_language(lang: str):
    """设置语言快捷函数"""
    _i18n.set_language(lang)
```

## 前端国际化（react-i18next）

```tsx
// frontend/src/i18n/config.ts
import i18n from 'i18next'
import { initReactI18next } from 'i18next/react-i18next'
import LanguageDetector from 'i18next-browser-languagedetector'
import HttpApi from 'i18next-http-backend'
import en from './locales/en.json'
import zh from './locales/zh.json'

const resources = {
  en: { translation: en },
  zh: { translation: zh },
}

i18n
  .use(HttpApi)
  .use(LanguageDetector)
  .use(initReactI18next)
  .init({
    resources,
    fallbackLng: 'zh',
    lng: 'zh',
    interpolation: {
      escapeValue: false,
    },
  })

export default i18n
```

## 语言切换组件

```tsx
// frontend/src/components/common/LanguageSwitcher.tsx
import { useTranslation } from 'react-i18next'
import { Button } from '@/components/ui/button'
import {
  DropdownMenu,
  DropdownMenuContent,
  DropdownMenuItem,
  DropdownMenuTrigger,
} from '@/components/ui/dropdown-menu'
import { Globe } from 'lucide-react'

const LANGUAGES = [
  { code: 'zh', name: '简体中文', flag: '🇨🇳' },
  { code: 'en', name: 'English', flag: '🇺🇸' },
  { code: 'ja', name: '日本語', flag: '🇯🇵' },
]

export function LanguageSwitcher() {
  const { i18n, t } = useTranslation()

  const changeLanguage = (lang: string) => {
    i18n.changeLanguage(lang)

    // 同步到后端
    fetch('/api/user/language', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ language: lang }),
    })
  }

  const currentLang = LANGUAGES.find(l => l.code === i18n.language) || LANGUAGES[0]

  return (
    <DropdownMenu>
      <DropdownMenuTrigger asChild>
        <Button variant="outline" size="sm">
          <Globe className="h-4 w-4 mr-2" />
          {currentLang.flag} {currentLang.name}
        </Button>
      </DropdownMenuTrigger>
      <DropdownMenuContent>
        {LANGUAGES.map((lang) => (
          <DropdownMenuItem
            key={lang.code}
            onClick={() => changeLanguage(lang.code)}
          >
            <span className="mr-2">{lang.flag}</span>
            {lang.name}
          </DropdownMenuItem>
        ))}
      </DropdownMenuContent>
    </DropdownMenu>
  )
}
```

## API 中间件

```python
# app/api/deps/i18n.py
from fastapi import Request, Header
from typing import Optional

from app.core.i18n import _i18n

async def get_language(
    request: Request,
    accept_language: Optional[str] = Header(None),
) -> str:
    """从请求中获取语言偏好"""
    # 优先使用用户设置的语言
    if hasattr(request.state, 'user') and request.state.user:
        user_lang = request.state.user.language
        if user_lang:
            return user_lang

    # 其次使用 Accept-Language 头
    if accept_language:
        # 解析 Accept-Language
        langs = accept_language.split(',')
        for lang in langs:
            lang_code = lang.split(';')[0].strip().lower()
            if lang_code in _i18n.translations:
                return lang_code

    # 默认中文
    return 'zh'
```
