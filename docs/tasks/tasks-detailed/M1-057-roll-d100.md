```python
import random

class DiceResult:

    raw_roll: int       # 原始掷骰值
    modifier: int        # 修正值
    final_value: int    # 最终值 = raw + modifier
```

```python
def roll_d100(modifier: int = 0) -> DiceResult:
    """掷一个 d100，返回 DiceResult"""
    raw_roll = random.randint(1, 100)
    final_value = raw_roll + modifier
    return DiceResult(
        dice_type=100,
        raw_roll=raw_roll,
        modifier=modifier,
        final_value=final_value
    )
```

```python
from app.core.dice import roll_d100

def test_roll_d100_range():
    """测试 d100 返回值在 1-100 之间"""
    for _ in range(100):
        result = roll_d100()
        assert 1 <= result.raw_roll <= 100

def test_roll_d100_with_modifier():
    """测试修正值正确应用"""
    result = roll_d100(modifier=10)
    assert result.final_value == result.raw_roll + 10
```
